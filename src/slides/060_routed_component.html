Inputs carry route data 

                    
                        @Component({
                            selector: 'app',
                            template: `
                                <ul>

                                    <li *ngFor="let route of routes">

                                        <button
                                            (click)="go(route)">{{ route }}
                                        </button>
                                </li>

                                </ul>`,
                            directives: [ROUTER_DIRECTIVES, Dumb]
                        })
                        @RouteConfig([
                            { path: '/Dumb', as: 'Dumb', component: Dumb },
                            { path: '/Smart', as: 'Smart', component: Smart }
                        ])
                        class App {
                            static get parameters () { return [[Router]]; }
                            constructor(router) {
                                this.router = router;
                                this.routes = ['Dumb', 'Smart']
                            }
                            go (to) { this.router.navigate(['/' + to]); }
                        }
