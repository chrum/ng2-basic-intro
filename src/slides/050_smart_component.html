States and services aware

Contains other components, renders them 

                    
                        import { Component } from '@angular/core';
                        import { NumberedRow } from './numbered-row';
                        import { ListData } from './list-data';
                    
                
                    
                        @Component({
                            selector: 'numbered-list',
                            directives: [NumberedRow],
                            template: `
                            <numbered-row

                            *ngFor="let item of items; let i = index"

                            [number]="(i + 1)"

                            [text]="item"></numbered-row>
                            `
                        })
                    
                
                    
                        export default class NumberedList {
                            static get parameters () {
                                  return [[ListData]];
                            }
                            constructor (listData) {
                                 this.items = listData.getData();
                            }
                        }
                    
                
                        
                            
                            <numbered-list></numbered-list>
